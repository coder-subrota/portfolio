Algorithm Implementation

Bellman-Ford Shortest Path

I implemented the Bellman-Ford algorithm in C++ to find the shortest paths from a single source vertex to all other vertices in a weighted graph. This implementation handles negative edge weights and is a core part of my graph theory studies.

Time Complexity O(V * E)
Space Complexity O(V)
  • Single Source Shortest Path
  • Negative Edge Weight Support
  • Edge List Representation
bellman_ford.cpp 

#include <bits/stdc++.h>
using namespace std;

class Edge {
public:
    int u, v, c;
    Edge(int u, int v, int c) {
        this->u = u;
        this->v = v;
        this->c = c;
    }
};

int main() {
    int n, e;
    cin >> n >> e;
    vector<Edge> EdgeList;
    while (e--) {
        int u, v, c;
        cin >> u >> v >> c;
        EdgeList.push_back(Edge(u, v, c));
    }

    // Algorithm Initialization
    int dis[n];
    for (int i = 0; i < n; i++) dis[i] = INT_MAX;
    dis[0] = 0;

    // Relaxation Process
    for (int i = 1; i <= n - 1; i++) {
        for (Edge ed : EdgeList) {
            if (dis[ed.u] < INT_MAX && dis[ed.u] + ed.c < dis[ed.v]) {
                dis[ed.v] = dis[ed.u] + ed.c;
            }
        }
    }
    return 0;
}
